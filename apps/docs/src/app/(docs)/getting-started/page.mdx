# Installation

This guide walks you from a fresh project to a working payment session in the sandbox environment.

## Install the package

```bash
npm install @better-sslcommerz/sdk
# or
pnpm add @better-sslcommerz/sdk
# or
yarn add @better-sslcommerz/sdk
```

## Set environment variables

Add your SSLCommerz credentials to your environment. For Next.js, put these in `.env.local`:

```bash
SSLCOMMERZ_STORE_ID=your_store_id
SSLCOMMERZ_STORE_PASSWD=your_store_password
```

<Note>
  These are **server-only** values. Never expose `SSLCOMMERZ_STORE_PASSWD` to
  the browser. In Next.js App Router, only use the client inside Server
  Components or Route Handlers.
</Note>

## Create a client

```ts
import { createSslcommerzClient } from "@better-sslcommerz/sdk";

export const sslcommerz = createSslcommerzClient({
  storeId: process.env.SSLCOMMERZ_STORE_ID!,
  storePasswd: process.env.SSLCOMMERZ_STORE_PASSWD!,
  environment: "sandbox", // switch to "live" for production
});
```

Create the client once and export it — it is stateless and safe to reuse across requests.

## Initiate your first payment

Call `client.core.createSession()` from a server Route Handler (or server action). The response contains a `GatewayPageURL` that you redirect the customer to.

```ts
// app/api/checkout/route.ts
import { NextResponse } from "next/server";
import { sslcommerz } from "@/lib/sslcommerz";

export async function POST(request: Request) {
  const body = await request.json();

  const response = await sslcommerz.core.createSession({
    total_amount: body.amount,
    currency: "BDT",
    tran_id: `TXN_${Date.now()}`, // must be unique per transaction
    success_url: "https://yoursite.com/payment/success",
    fail_url: "https://yoursite.com/payment/fail",
    cancel_url: "https://yoursite.com/payment/cancel",
    ipn_url: "https://yoursite.com/api/ipn",
    product_name: body.productName,
    product_category: "general",
    product_profile: "general",
    cus_name: body.customerName,
    cus_email: body.customerEmail,
    cus_add1: body.address,
    cus_city: body.city,
    cus_postcode: body.postcode,
    cus_country: "Bangladesh",
    cus_phone: body.phone,
    shipping_method: "NO",
    num_of_item: 1,
  });

  if (response.status !== "SUCCESS" || !response.GatewayPageURL) {
    return NextResponse.json(
      { error: response.failedreason ?? "Session creation failed" },
      { status: 400 },
    );
  }

  // Redirect the customer to the SSLCommerz payment page
  return NextResponse.json({ url: response.GatewayPageURL });
}
```

On the client side, redirect to the returned URL:

```ts
const res = await fetch("/api/checkout", {
  method: "POST",
  body: JSON.stringify(orderData),
  headers: { "Content-Type": "application/json" },
});
const { url } = await res.json();
window.location.href = url;
```

## Handle callbacks

SSLCommerz will POST to your `success_url`, `fail_url`, and `cancel_url` after the payment attempt. Always validate the result using the [Validate Order](/core/validate-order) endpoint — do **not** trust the redirect alone.

```ts
// app/payment/success/route.ts
import { NextResponse } from "next/server";
import { sslcommerz } from "@/lib/sslcommerz";

export async function POST(request: Request) {
  const form = await request.formData();
  const valId = form.get("val_id") as string;

  const validation = await sslcommerz.core.validateOrder({ val_id: valId });

  if (validation.status !== "VALID" && validation.status !== "VALIDATED") {
    return NextResponse.redirect(new URL("/payment/failed", request.url));
  }

  // Mark order as paid in your database
  // Always cross-check validation.amount and validation.tran_id

  return NextResponse.redirect(new URL("/payment/thank-you", request.url));
}
```
